                var Chain = REO.chain({
                    module : 'user_next',
                    val: [22,3,5] // mes1 = 22 ,mes2 = 3, mes3 = 5
                });

                // 定义链式调用
                var getvalue = Chain.then(function ( api, mes1 ) {
                    api( mes1 + 3 );
                })
                .then(function  ( api , mes2, massages ) {
                    api( mes2 +  Number(input.value)  );
                })
                .then(function  ( api , mes3, massages ) {
                    api( mes3 + massages );
                });

                // 输出所有
                var getAll = getvalue.all();  // => 25,15,20

                // 或者（使用模块名输出）
